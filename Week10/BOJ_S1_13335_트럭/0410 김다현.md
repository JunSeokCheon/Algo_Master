```python
n,l,w = map(int,input().split())
lst = list(map(int,input().split()))

bridge = [0]*l
weight, time = 0,0
while bridge:
    time +=1 #조건 상관없이 시간은 흐름
    bridge.pop(0)
    if lst: #트럭이 존재할 때만 다리에 append
        if sum(bridge) + lst[0] <= w:
            bridge.append(lst.pop(0))
        else:
            bridge.append(0) #다리에 올리지 못한 경우
print(time) 
```
