import sys
from collections import deque
input = sys.stdin.readline

    
def check():
    for num_c,c in enumerate(arr):
        for num_r, r in enumerate(c):
            if r=='O':
                q.append((num_c,num_r))


def fill():
    global arr
    arr = [['O'] for _ in range(c)]

    
def bomb():
    while q:
        y,x = q.popleft()
        arr[y][x]='.'
        for i in range(4):
            ny,nx = y+dy[i], x+dx[i]
            if 0<=nx<r and 0<=ny<c:
                arr[ny][nx]='.'

c,r,n = map(int, input().split())
    
dx=[-1,1,0,0]
dy= [0,0,-1,1]

arr=[]
for _ in range(c):
    arr.append(list(map(str, input().strip())))

n-=1
while n:
    q = deque()
    check()
    fill()
    n-=1
    if n==0:
        break
    bomb()
    n-=1
    
for i in arr:
    print(*i, sep='')
