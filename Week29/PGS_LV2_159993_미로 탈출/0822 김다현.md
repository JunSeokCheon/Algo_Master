```python
from collections import deque
q= deque()
arr= []
def solution(maps):
    dy = [0, 1, -1, 0]
    dx = [1, 0, 0, -1]

    n,m = len(maps), len(maps[0])
    for col,i in enumerate(maps):
        arr.append(list(i))
        for row,j in enumerate(i):
            if j=='S':
                s= (col,row)
            if j=='L':
                l = (col,row)
            if j=='E':
                e = (col,row)
    def bfs(start,end):
        visited = [[0]*m for _ in range(n)]
        x,y= start
        q.append((x,y))
        visited[x][y]=1

        while q:
            x,y= q.popleft()
            if arr[x][y]==end:
                return visited[x][y]-1
            for i in range(4):
                nx= x+dx[i]
                ny= y+dy[i]
                if 0<=nx<n and 0<=ny<m and arr[nx][ny]!= 'X' and visited[nx][ny]==0:
                    visited[nx][ny]=visited[x][y]+1
                    q.append((nx,ny))
        return -1
    path1 = bfs(s, 'L')	# 시작 지점 --> 레버
    path2 = bfs(l, 'E') # 레버 --> 출구
    
    # 둘다 -1 이 아니라면 탈출할 수 있음
    if path1 != -1 and path2 != -1:
        return path1 + path2
        
   	# 둘중 하나라도 -1 이면 탈출할 수 없음
    return -1
```
