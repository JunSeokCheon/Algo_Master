``` 
1. 맨 앞글자를 pop하기 위해 입력받은 문자열을 deque로 설정
2. 각 단계별로 해당되는 문자가 있을 경우 popleft()
3. 마지막 까지 진행 후 deque가 비어있으면 Infected! 비어있지 않으면 Good 출력
  
```

```python
# 방법 1. 시간초과

import sys
from copy import deepcopy

input = sys.stdin.readline

n = int(input())
arr = list(map(int, input().split()))

result = -100000

for i in range(n):
    temp = deepcopy(arr)
    sum_ = temp[i:]
    while sum_ and temp:
        value = sum(sum_)
        if result < value:
            result = value
        sum_.pop()


print(result)


# 방법 2. 
import sys

input = sys.stdin.readline

n = int(input())
arr = list(map(int, input().split()))
result = [0] * n
result[0] = arr[0]

for i in range(1, n):
    result[i] = max(arr[i], arr[i] + result[i - 1])

print(max(result))
```
