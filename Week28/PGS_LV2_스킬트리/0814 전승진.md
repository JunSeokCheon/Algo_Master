```python
def solution(skill, skill_trees):
    answer = 0
    for i in skill_trees:
        check=0
        check_tree=""
        for j in i:
            if j in skill:
                if j not in check_tree:
                    check_tree+=j
                    
                    if skill.index(j) == check_tree.index(j):
                    
                        check+=1
                    else:
                        check_tree=check_tree[:-1]
            else:
                check+=1
        if check == len(i):
            answer+=1
 
    return answer
```
