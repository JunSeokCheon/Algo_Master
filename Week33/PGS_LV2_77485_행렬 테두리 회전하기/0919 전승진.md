from collections import deque
def solution(rows, columns, queries):
    answer = []
    n= rows
    m = columns
    arr=[[0]*m for _ in range(n)]
    for i in range(n):
        for j in range(m):
            arr[i][j]=i*m+j+1
    #print(arr)
    for rotation in queries:
        #tmp=deque()
        
       
        a,b,c,d  = rotation
        ##왼쪽위 오른쪽아래
        a=a-1
        b=b-1
        c=c-1
        d=d-1
        ###가장 작은수 확인
        #print(a,b,c,d)
        ###돌리기
            
        tmp_2=deque()
            #맨위 마지막은 오른쪽에서 받기
        for e in range(b,d):
            tmp_2.append(arr[a][e])
            
            #오른쪽
        for e in range(a,c):
            tmp_2.append(arr[e][d])
            #아래
        for e in range(d,b,-1):
            tmp_2.append(arr[c][e])
            #왼쪽
        for e in range(c,a,-1):
            tmp_2.append(arr[e][b])
        #print(tmp_2)
        answer.append(min(tmp_2))
       
       
        tmp_2.rotate(1)
        
        for e in range(b,d):
            num = e-b
            arr[a][e]=tmp_2[num]
            #print(tmp_2[num])
            #오른쪽
        for e in range(a,c):
            num = e-a
            arr[e][d]=tmp_2[num+d-b]
            #아래
            #print(tmp_2[num+d-b-1])
        for e in range(d,b,-1):
            num = d-e
          
            arr[c][e]=tmp_2[num+c-a+d-b]
            #print(tmp_2[num+c-a+d-b])
          
            #왼쪽
        for e in range(c,a,-1):
            num = c-e
            #print(num+c-a+d-b+d-b)
     
            arr[e][b]=tmp_2[num+c-a+d-b+d-b]
            #print(tmp_2[num+c-a+d-b+d-b])
        #print(arr)
           
        
    return answer
