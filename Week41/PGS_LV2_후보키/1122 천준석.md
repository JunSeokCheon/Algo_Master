

```python
from itertools import combinations

def solution(relation):
    row = len(relation)
    col = len(relation[0])
    
    combi = []
    for i in range(1, col+1):
        # 열의 모든 조합을 구해서 처리
        combi.extend(combinations(range(col), i))
    
    result = []
    for i in combi:
        temp = [item[key] for key in i for item in relation]
        
        # 유일성과 최소성 만족시키는 로직
        if len(set(temp)) == row:
            good = True
            
            for x in result:
                if set(x).issubset(set(i)):
                    good = False
                    break
            
            # 만족시키면 결과에 추가
            if good:
                result.append(i)
    print(len(result))
    return len(result)
```
