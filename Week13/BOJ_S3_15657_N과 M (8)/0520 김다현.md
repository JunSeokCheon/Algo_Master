import sys
input = sys.stdin.readline

n,m = map(int, input().split())
num = sorted(list(map(int, input().strip().split())))
result=[]
def back(answer,start):
    global result
    if len(answer)==m:
        result.append(answer[:])
        return
    for i in range(start, n):
        answer.append(num[i])
        back(answer,i)
        answer.pop()
back([],0)
for i in sorted(result):
    print(*i)
