import sys
input = sys.stdin.readline
n,s=map(int,input().split())

lst = []
arr= list(input().split())


def dfs(path, cnt):
    global lst
    print(lst)
    if cnt >= s:
        lst.append([path])
        return
    for i in range( n):
        dfs(path+arr[i], cnt+1)

dfs('',0)

for i in sorted(lst):
    print(*i)
