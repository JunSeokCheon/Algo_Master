import sys
input = sys.stdin.readline
n= int(input())
stack =[]

for i in range(n):
    stack.append(list(input().strip()))

def dfs(elements):
    #리프 노트일때 결과 추가
    if len(elements)==0:
        result.append(''.join(prev_elements[:]))
        return #Q.이거 없어도됨 왜..?
    #순열 생성 재귀 호출

    for e in elements:
        next_elements = elements[:]
        next_elements.remove(e) #자기자신 제거
        
        prev_elements.append(e)
        # print(f'elements : {elements}')
        # print(f'next : {next_elements}')
        # print(f'prev : {prev_elements}')
        dfs(next_elements)
        # print('====out recursion')
        prev_elements.pop()
for st in stack:
