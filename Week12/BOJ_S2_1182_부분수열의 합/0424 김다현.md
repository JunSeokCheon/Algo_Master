import sys
input = sys.stdin.readline
n,t = map(int,input().split())
lst = list(map(int,input().split()))
lst= sorted(lst)
result =0 
def dfs(csum, idx, cnt, path):
    global result
    #n을 넘으면 종료
    if cnt > n:
        return
    if csum==t and path!='':
        result+=1
    for i in range(idx, len(lst)):
        dfs(csum+lst[i], i+1, cnt+1, path+str(lst[i]))

dfs(0,0,0,'')
print(result)
