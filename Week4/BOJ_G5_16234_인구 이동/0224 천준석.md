# BOJ_G5_16234_인구 이동 [2023-02-24]
https://www.acmicpc.net/problem/16234

<접근법>
``` 
0. bfs를 한번 돌 때마다 방문 여부에 따라 인구 통합을 한다. (상하좌우를 확장할 que와 연합을 저장한 리스트 사용)
1. 연합 수가 2개 이상일 때 인구 통합 및 FLAG 변경
2. FLAG가 변경 되지 않으면 무한루프 탈출 후 결과 출력

```


```python
import sys
from collections import deque

dx = [-1, 1, 0, 0]
dy = [0, 0, -1, 1]
N, L, R = map(int, sys.stdin.readline().split())
ground = [list(map(int, sys.stdin.readline().split())) for _ in range(N)]

def bfs(i,j):
    # 상하좌우 확장할 que1
    que = deque()
    que.append((i,j))
    # 인구를 저장할 que2
    que2 = []
    que2.append((i,j))
    
    while que:
        x, y = que.popleft()
        # 상하좌우 이동
        for i in range(4):
            nx = x + dx[i]
            ny = y + dy[i]

            if nx < 0 or nx >= N or ny < 0 or ny >= N:
                continue
            
            # 아직 방문하지 않은 나라이고, 해당 나라의 인구가 L명 이상, R명 이하라면
            if not visited[nx][ny]:
                if L <= abs(ground[x][y] - ground[nx][ny]) <= R:
                    # 방문 표시와 확장 QUE, 연합 인구 QUE에 좌표 추가
                    visited[nx][ny] = True
                    que.append((nx, ny))
                    que2.append((nx, ny))
    return que2

answer = 0
while True:
    flag = False
    visited = [[False]*N for _ in range(N)]
    for i in range(N):
        for j in range(N):
            if not visited[i][j]:
                visited[i][j] = True
                group = bfs(i,j)
                # 연합안에 나라의 개수가 2개 이상일 때만 평균 인구 계산
                if len(group) > 1:
                    flag = True
                    population = sum([ground[x][y] for x, y in group]) // len(group)
                    for x, y in group:
                        ground[x][y] = population
    # 더 이상 인구이동이 발생하지 않는다면 break
    if not flag:
        break
    # 인구 이동시 카운트 추가
    else:
        answer += 1
print(answer)
```
