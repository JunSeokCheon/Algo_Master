while True:
    try:
        s,t = input().split(' ')
        flag = True
        for i in range(len(s)-1):
            if s[i] not in t:
                flag = False
                break
            if s[i+1] not in t[t.index(s[i]):]:
                flag = False
                break
        answer ='Yes' if flag == True else 'No'
        print(answer)
    except:
        exit(0)
